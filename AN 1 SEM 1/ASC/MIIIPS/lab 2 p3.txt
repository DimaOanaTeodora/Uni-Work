.data 
n: .space 4
v: .space 400
sp: .asciiz " "
.text
main:
	li $v0, 5
	syscall
	move $t0, $v0
	
	li $t1,0
	li $t2,0
citire_vector:
	bge $t1, $t0, exit1
	li $v0, 5
	syscall
	sw $v0, v($t2)
	addi $t1,1
	addi $t2,4
j citire_vector
exit1:
	li $t1,0
	li $t2,0
afisare_val:
	bge $t1, $t0, exit
	lw $t3,v($t2)
	rem $t4,$t3,2
	beq $t4,$zero,afisare
	
cont: addi $t1,1
	addi $t2,4

	
j afisare_val
afisare:
	lw $a0,v($t2)
	li $v0,1
	syscall
	la $a0,sp
	li $v0,4
	syscall
j cont
	
exit:
	li $v0,10
	syscall
